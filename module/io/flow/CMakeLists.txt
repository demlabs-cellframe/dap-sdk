cmake_minimum_required(VERSION 3.10)
project(dap_io_flow C)

include(${CMAKE_CURRENT_SOURCE_DIR}/../../../cmake/LibraryHelpers.cmake)

file(GLOB DAP_IO_FLOW_SOURCES *.c)
file(GLOB DAP_IO_FLOW_HEADERS include/*.h)

if(LINUX OR ANDROID)
    file(GLOB PLAT_SRCS linux/*.c)
    file(GLOB PLAT_HDRS linux/*.h)
    list(APPEND DAP_IO_FLOW_SOURCES ${PLAT_SRCS})
    list(APPEND DAP_IO_FLOW_HEADERS ${PLAT_HDRS})
elseif(APPLE)
    file(GLOB PLAT_SRCS darwin/*.c)
    file(GLOB PLAT_HDRS darwin/*.h)
    list(APPEND DAP_IO_FLOW_SOURCES ${PLAT_SRCS})
    list(APPEND DAP_IO_FLOW_HEADERS ${PLAT_HDRS})
elseif(BSD)
    file(GLOB PLAT_SRCS bsd/*.c)
    file(GLOB PLAT_HDRS bsd/*.h)
    list(APPEND DAP_IO_FLOW_SOURCES ${PLAT_SRCS})
    list(APPEND DAP_IO_FLOW_HEADERS ${PLAT_HDRS})
elseif(WIN32)
    file(GLOB PLAT_SRCS windows/*.c)
    file(GLOB PLAT_HDRS windows/*.h)
    list(APPEND DAP_IO_FLOW_SOURCES ${PLAT_SRCS})
    list(APPEND DAP_IO_FLOW_HEADERS ${PLAT_HDRS})
endif()

create_object_library(${PROJECT_NAME} DAP_INTERNAL_MODULES ${DAP_IO_FLOW_SOURCES} HEADERS ${DAP_IO_FLOW_HEADERS})

dap_link_libraries(${PROJECT_NAME} INTERFACE dap_core dap_io dap_crypto dap_hash dap_hash_table)

target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)
if(LINUX OR ANDROID)
    target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/linux)
elseif(APPLE)
    target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/darwin)
elseif(BSD)
    target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/bsd)
elseif(WIN32)
    target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/windows)
endif()

message("[+] Module 'io_flow' (5.8 datagram reactor, context_queue, thread_pool)")
